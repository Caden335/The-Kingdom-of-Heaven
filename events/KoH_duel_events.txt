	title = KoH_duels.3.t

	is_triggered_only = yes
	
	theme = war
	
	left_portrait = {
		character = scope:combatant_1
		animation = personality_bold
	}
	right_portrait = {
		character = scope:combatant_2
		animation = personality_bold
	}
	
	option = {
		name = KoH_duels.3.a
		duel = {
			skill = prowess
			target = scope:combatant_2
			value = 10
			10 = {
				compare_modifier = {
					value = scope:duel_value
				}
				trigger_event = {
					id = KoH_duels.4 #I won
					days = 1
				}
			}
			10 = {
				trigger_event = {
					id = KoH_duels.5 #I lost
					days = 1
				}
			}
		}
	}
}
